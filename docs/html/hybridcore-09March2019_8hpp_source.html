307 template<typename Function>
308 Eigen::VectorXd HybridCore::interpolate(const Function& f, size_t doe) const {
309  Eigen::VectorXd XTF = Eigen::VectorXd::Zero(_ntotal_dofs);
310 
311  for (size_t iT = 0; iT < _mesh_ptr->n_cells(); iT++) {
312 
313  // Mass matrix in cell
314  std::vector<HybridCore::qrule> quadT = cell_qrule(iT, doe);
315  std::vector<Eigen::VectorXd> phi_quadT = basis_quad('T', iT, quadT, _nlocal_cell_dofs);
316  Eigen::MatrixXd MT = gram_matrix(phi_quadT, phi_quadT, _nlocal_cell_dofs, _nlocal_cell_dofs, quadT, true);
317 
318  // Vector of integrals of f against basis functions
319  Eigen::VectorXd bT = Eigen::VectorXd::Zero(_nlocal_cell_dofs);
320  for (size_t i = 0; i < _nlocal_cell_dofs; i++) {
321  const auto& phi_i = cell_basis(iT, i);
322  bT(i) += integrate_over_cell(iT, [&phi_i, &f](double x, double y) {
323  return phi_i(x,y) * f(x,y);
324  });
325  }
326 
327  // Vector of coefficients (on cell basis functions) of the L2(T) projection of f
328  Eigen::VectorXd UT = MT.ldlt().solve(bT);
329 
330  // Fill in the complete vector of DOFs
331  size_t offset_T = iT * _nlocal_cell_dofs;
332  XTF.segment(offset_T, _nlocal_cell_dofs) = UT;
333 
334  for (size_t ilF = 0; ilF < _mesh_ptr->cell(iT)->n_edges(); ilF++) {
335  size_t iF = _mesh_ptr->cell(iT)->edge(ilF)->global_index();
336 
337  // Mass matrix on face
338  std::vector<HybridCore::qrule> quadF = edge_qrule(iF, doe);
339  std::vector<Eigen::VectorXd> phiF_quadF = basis_quad('F', iF, quadF, _nlocal_edge_dofs);
340  Eigen::MatrixXd MF = gram_matrix(phiF_quadF, phiF_quadF, _nlocal_edge_dofs, _nlocal_edge_dofs, quadF, true);
341 
342  // Vector of integral of f against face basis functions
343  Eigen::VectorXd bF = Eigen::VectorXd::Zero(_nlocal_edge_dofs);
344  for (size_t i = 0; i < _nlocal_edge_dofs; i++) {
345  const auto& phi_i = edge_basis(iF, i);
346  bF(i) += integrate_over_edge(iF, [&phi_i, &f](double x, double y) {
347  return phi_i(x,y) * f(x,y);
348  });
349  }
350 
351  // Vector of coefficients (on cell basis functions) of the L2(F) projection of f
352  Eigen::VectorXd UF = MF.ldlt().solve(bF);
353 
354  // Fill in he complete vector of DOFs
355  XTF.segment(_ntotal_cell_dofs + iF * _nlocal_edge_dofs, _nlocal_edge_dofs) = UF;
356  }
357 
358  // Special case of L=-1, we replace the previously computed cell value with the proper average of face values
359  if ( _L==-1 ){
360  Eigen::VectorXd barycoefT = compute_weights(iT);
361  XTF(iT) = 0;
362  for (size_t ilF = 0; ilF < _mesh_ptr->cell(iT)->n_edges(); ilF++) {
363  size_t iF = _mesh_ptr->cell(iT)->edge(ilF)->global_index();
364  XTF(iT) += barycoefT(ilF) * XTF(_ntotal_cell_dofs + iF);
365  }
366  }
367 
368  }
369 
370  return XTF;
371 }
