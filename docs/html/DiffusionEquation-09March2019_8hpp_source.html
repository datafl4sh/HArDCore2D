85 Eigen::VectorXd DiffusionEquation::solve(HybridCore &hho) {
86 
87  boost::timer timer; // Time the matrix assembly
88  boost::timer timer2; // Time the matrix assembly
89  const auto mesh = hho.get_mesh_ptr();
90 
91  // System matrix
92  Eigen::SparseMatrix<double> A(hho.ntotal_edge_dofs(), hho.ntotal_edge_dofs());
93  // If static condensation (L>=0): matrix to recover cell unknowns
94  // If barycentric elimination (L=-1): matrix to recover cell unknowns
95  Eigen::SparseMatrix<double> A_sc_be(hho.ntotal_cell_dofs(), hho.ntotal_edge_dofs());
96 
97  Eigen::VectorXd B = Eigen::VectorXd::Zero(hho.ntotal_edge_dofs());
98  Eigen::VectorXd B_sc = Eigen::VectorXd::Zero(hho.ntotal_cell_dofs());
99 
100  //------------------------------------------------------------------------------
101  // Assemble local contributions
102 
103  // Global quadrature rule for the cells
104 
105  Eigen::VectorXd cell_quadrature = Eigen::VectorXd::Zero(hho.ntotal_cell_dofs());
106 
107 
108  std::vector<Eigen::Triplet<double>> triplets;
109  std::vector<Eigen::Triplet<double>> triplets_sc_be; // Triplets to create A_sc or A_be
110 
111  auto total_measure = 0.0;
112 
113 double ttt = 0.0;
114 double ttt2 = 0.0;
115 double ttt3 = 0.0;
116 double ttt4 = 0.0;
117 
118 
119 
120  for (size_t iT = 0; iT < mesh->n_cells(); iT++) {
121  Cell2D* iCell = mesh->cell(iT);
122 
123 timer2.restart();
124  total_measure += iCell->get_cell_area();
125 
126  // Total number of face degrees of freedom local to this cell (adjacent faces to the cell)
127  size_t nlocal_edges = iCell->n_edges();
128  size_t edge_dofs = nlocal_edges * hho.nlocal_edge_dofs();
129  // Compute local operators
130 
131  Eigen::VectorXd bT = load_operator(hho, iT);
132  Eigen::MatrixXd aT = diffusion_operator(hho, iT, ttt, ttt2, ttt3, ttt4);
133 
134 
135 
136  // Local matrix and right-hand side on the face unknowns
137  Eigen::MatrixXd MatF = Eigen::MatrixXd::Zero(edge_dofs,edge_dofs);
138  Eigen::VectorXd bF = Eigen::VectorXd::Zero(edge_dofs);
139 
140  if (hho.L()>=0) {
141  // STATIC CONDENSATION OF ELEMENT UNKNOWNS
142 
143  // Perform static condensation
144  Eigen::MatrixXd ATT = aT.topLeftCorner(hho.nlocal_cell_dofs(), hho.nlocal_cell_dofs());
145  Eigen::MatrixXd ATF = aT.topRightCorner(hho.nlocal_cell_dofs(), edge_dofs);
146  Eigen::MatrixXd AFF = aT.bottomRightCorner(edge_dofs, edge_dofs);
147 
148  Eigen::PartialPivLU<Eigen::MatrixXd> invATT;
149  invATT.compute(ATT);
150 
151  Eigen::MatrixXd invATT_ATF = invATT.solve(ATF);
152  Eigen::VectorXd invATT_bTcell = invATT.solve(bT.head(hho.nlocal_cell_dofs()));
153  MatF = AFF - ATF.transpose() * invATT_ATF;
154 
155  bF = bT.tail(edge_dofs) - ATF.transpose() * invATT_bTcell;
156 
157  for (size_t i =0; i < size_t(bF.size()); i++){
158  if(std::isnan(bF[i])){
159  std::cout << "The vector bF is NaN"<<std::endl;
160  }
161  }
162 
163  // Assemble static condensation operator
164  B_sc.segment(iT * hho.nlocal_cell_dofs(), hho.nlocal_cell_dofs()) = invATT_bTcell;
165  for (size_t i = 0; i < hho.nlocal_cell_dofs(); i++) {
166  for (size_t jlF = 0; jlF < nlocal_edges; jlF++) {
167  const size_t jF = iCell->edge(jlF)->global_index();
168  for (size_t jk = 0; jk < hho.nlocal_edge_dofs(); jk++) {
169  const size_t jLocal = jlF * hho.nlocal_edge_dofs() + jk;
170  const size_t jGlobal = jF * hho.nlocal_edge_dofs() + jk;
171  triplets_sc_be.emplace_back(iT * hho.nlocal_cell_dofs() + i, jGlobal, invATT_ATF(i, jLocal));
172  }
173  }
174  }
175  } else {
176  // BARYCENTRIC ELIMINATION OF ELEMENT UNKNOWNS
177  // Create reduction matrix: 1+nlocal_edges * nlocal_edges matrix with the coefficients on the first row, and the identity below. When multiplied by the face unknowns, return cell and face unknowns
178  Eigen::MatrixXd red_matT = Eigen::MatrixXd::Zero(1+nlocal_edges,nlocal_edges);
179  red_matT.row(0) = hho.compute_weights(iT);
180  red_matT.bottomRightCorner(nlocal_edges,nlocal_edges) = Eigen::MatrixXd::Identity(nlocal_edges,nlocal_edges);
181 
182  bF = red_matT.transpose() * bT;
183  MatF = red_matT.transpose() * aT * red_matT;
184 
185  // To recover cell unknown
186  for (size_t jlF = 0; jlF < nlocal_edges; jlF++) {
187  const size_t jF = iCell->edge(jlF)->global_index();
188  const size_t jGlobal = jF * hho.nlocal_edge_dofs();
189  triplets_sc_be.emplace_back(iT, jGlobal, red_matT(0,jlF));
190  }
191 
192  }
193 
194 
195  // Assemble local contribution into global matrix
196  for (size_t ilF = 0; ilF < nlocal_edges; ilF++) {
197  const size_t iF = iCell->edge(ilF)->global_index();
198  for (size_t ik = 0; ik < hho.nlocal_edge_dofs(); ik++) {
199  const size_t iLocal = ilF * hho.nlocal_edge_dofs() + ik;
200  const size_t iGlobal = iF * hho.nlocal_edge_dofs() + ik;
201  for (size_t jlF = 0; jlF < nlocal_edges; jlF++) {
202  const size_t jF = iCell->edge(jlF)->global_index();
203  for (size_t jk = 0; jk < hho.nlocal_edge_dofs(); jk++) {
204  const size_t jLocal = jlF * hho.nlocal_edge_dofs() + jk;
205  const size_t jGlobal = jF * hho.nlocal_edge_dofs() + jk;
206  triplets.emplace_back(iGlobal, jGlobal, MatF(iLocal, jLocal));
207  }
208  }
209  B(iGlobal) += bF(iLocal);
210  }
211  }
212 
213  // Record cell quadrature
214  if (BC==1){
215  for (size_t i = 0; i < hho.nlocal_cell_dofs(); i++) {
216  const auto& phi_i = hho.cell_basis(iT, i);
217  cell_quadrature(iT * hho.nlocal_cell_dofs() + i) = hho.integrate_over_cell(iT, phi_i);
218  }
219  }
220 ttt+=timer2.elapsed();
221 
222  }
223  if (BC==0){
224  // Dirichlet boundary conditions: the equations for boundary faces are replaced
225  // with MF UF = (int of phi_i u)_i. This forces UF to represent the coefficients
226  // of the L2(F) projection of u on the basis (phi_i)_i
227  std::vector<size_t> bedges = mesh->b_edges(); //List of boundary global indices
228  for (size_t ibF = 0; ibF < mesh->n_bedges(); ibF++){
229  size_t iF = bedges[ibF];
230  // Mass matrix and boundary values
231  std::vector<HybridCore::qrule> quadF = hho.edge_qrule(iF, 2*hho.K());
232  std::vector<Eigen::VectorXd> phiF_quadF = hho.basis_quad('F', iF, quadF, hho.nlocal_edge_dofs());
233  Eigen::MatrixXd MF = hho.gram_matrix(phiF_quadF, phiF_quadF, hho.nlocal_edge_dofs(), hho.nlocal_edge_dofs(), quadF, true);
234  for (size_t j = 0; j < hho.nlocal_edge_dofs(); j++){
235  // Erase row in matrix and replace
236  size_t jGlobal = iF * hho.nlocal_edge_dofs() + j;
237  triplets.erase(std::remove_if(std::begin(triplets), std::end(triplets),
238  [&jGlobal](const auto& x) { return (x.row() == (int) jGlobal); }), std::end(triplets));
239  for (size_t k = 0; k < hho.nlocal_edge_dofs(); k++){
240  size_t kGlobal = iF * hho.nlocal_edge_dofs() + k;
241  triplets.emplace_back(jGlobal, kGlobal, MF(j,k));
242  }
243  // Replace boundary value
244  const auto& phi_j = hho.edge_basis(iF, j);
245  B(jGlobal) = hho.integrate_over_edge(iF, [this, &phi_j](double x, double y) {
246  return phi_j(x,y) * exact_solution(x,y); });
247  }
248 
249  }
250  } else if (BC==1){
251  // Remove a row in the matrix and fix the first degree of freedom
252  triplets.erase(std::remove_if(std::begin(triplets), std::end(triplets),
253  [](const auto& x) { return (x.row() == 0); }), std::end(triplets));
254  triplets.emplace_back(0, 0, 1);
255  B(0) = 0;
256  }
257  // Assemble the linear system
258  A.setFromTriplets(std::begin(triplets), std::end(triplets));
259  A_sc_be.setFromTriplets(std::begin(triplets_sc_be), std::end(triplets_sc_be));
260 
261 
262 
263  assembly_time = timer.elapsed(); // Record final assembly time
264  timer.restart(); // Time the solving
265 
266  // Solve the condensed system
267  Eigen::VectorXd xF = Eigen::VectorXd::Zero(hho.ntotal_edge_dofs());
268 // if (solver_type == "ma41") {
269 // Eigen::MA41<Eigen::SparseMatrix<double>, Eigen::VectorXd> solver;
270 // solver.analyzePattern(A);
271 // solver.factorize(A);
272 // xF = solver.solve(B);
273 // } else {
274  Eigen::BiCGSTAB<Eigen::SparseMatrix<double> > solver;
275  solver.compute(A);
276  xF = solver.solve(B);
277  std::cout << "#iterations: " << solver.iterations() << std::endl;
278  std::cout << "estimated error: " << solver.error() << std::endl;
279 // }
280  // Recover the cell unknowns
281  Eigen::VectorXd XTFh = Eigen::VectorXd::Zero(hho.ntotal_dofs());
282  XTFh.tail(hho.ntotal_edge_dofs()) = xF;
283 
284  if (hho.L()>=0) {
285  XTFh.head(hho.ntotal_cell_dofs()) = B_sc - A_sc_be * xF.head(hho.ntotal_edge_dofs());
286  } else {
287  XTFh.head(hho.ntotal_cell_dofs()) = A_sc_be * xF.head(hho.ntotal_edge_dofs());
288  }
289 
290  // Only Neumann: translate to get the proper average
291  if (BC==1){
292  // Compute average to translate
293  auto average = cell_quadrature.dot(XTFh.head(hho.ntotal_cell_dofs())) / total_measure;
294  auto average_exact_sol = hho.integrate_over_domain([&](auto x, auto y) {
295  return exact_solution(x,y);
296  }) / total_measure;
297 
298  // Translate the cells
299  for (size_t iT = 0; iT < mesh->n_cells(); iT++) {
300  XTFh(iT * hho.nlocal_cell_dofs()) += average_exact_sol - average;
301  }
302 
303  for (size_t iF = 0; iF < mesh->n_edges(); iF++) {
304  XTFh(hho.ntotal_cell_dofs() + iF * hho.nlocal_edge_dofs()) += average_exact_sol - average;
305  }
306  }
307 
308  solving_time = timer.elapsed(); // Record the final solving time
309 
310 std::cout << "time int ttt " << ttt << "\n";
311 std::cout << "time int ttt2 " << ttt2 << "\n";
312 std::cout << "time int ttt3 " << ttt3 << "\n";
313 std::cout << "time int ttt4 " << ttt4 << "\n";
314 
315  // Compute the error in the linear system
316  solving_error = (A * xF - B).norm();
317 
318  return XTFh;
319 }

321 Eigen::MatrixXd DiffusionEquation::diffusion_operator(HybridCore &hho, const size_t iT, double &ittt, double &ittt2, double &ittt3, double &ittt4) const {
322 
323 boost::timer timeint;
324 
325  const auto mesh = hho.get_mesh_ptr();
326 
327  // Total number of degrees of freedom local to this cell (cell and its adjacent faces)
328  size_t local_dofs = hho.nlocal_cell_dofs() + mesh->cell(iT)->n_edges() * hho.nlocal_edge_dofs();
329 
330  //------------------------------------------------------------------------------
331  // Initialisaton
332 
333  // Identity tensor
334  tensor_function_type identity = [&](double x, double y) {
335  return Eigen::Matrix2d::Identity();
336  };
337 
338  // Compute cell quadrature points and values of cell basis functions, gradients, and kappa*gradients at these points
339  std::vector<HybridCore::qrule> quadT = hho.cell_qrule(iT, hho.Ldeg()+hho.K()+1);
340  std::vector<Eigen::VectorXd> phi_quadT = hho.basis_quad('T', iT, quadT, hho.nhighorder_dofs());
341  std::vector<Eigen::MatrixXd> dphiT_quadT = hho.diff_grad_quad(iT, quadT, identity, hho.ngradient_dofs());
342  std::vector<Eigen::MatrixXd> kappa_dphiT_quadT = hho.diff_grad_quad(iT, quadT, kappa, hho.ngradient_dofs());
343 
344 
345 //timeint.restart();
346 //ittt += timeint.elapsed();
347 //timeint.restart();
348 
349  //------------------------------------------------------------------------------
350  // Compute gradient reconstruction
351 
352  // Stiffness matrix (kappa dphi_i,dphi_j)_T
353  Eigen::MatrixXd MG = hho.gram_matrix(dphiT_quadT, kappa_dphiT_quadT, hho.ngradient_dofs(), hho.ngradient_dofs(), quadT, true);
354 
355 //ittt2 += timeint.elapsed();
356 //timeint.restart();
357 
358  // mass matrix of (phi_i,phi_j)_T for phi_i up to degree L and phi_j up to degree K+1
359  Eigen::MatrixXd MTT = hho.gram_matrix(phi_quadT, phi_quadT, hho.nlocal_cell_dofs(), hho.nhighorder_dofs(), quadT, true);
360 
361 
362  // Right-hand side
363  Eigen::MatrixXd BG = Eigen::MatrixXd::Zero(hho.ngradient_dofs(), local_dofs);
364  BG.block(0, 1, hho.ngradient_dofs(), hho.nlocal_cell_dofs()-1) = MG.topLeftCorner(hho.ngradient_dofs(), hho.nlocal_cell_dofs()-1);
365 //ittt3 += timeint.elapsed();
366 
367 //timeint.restart();
368 
369  // Preparing matrices to store face mass matrices (for face and cell basis functions), and face-cell mass matrices
370  std::vector<Eigen::MatrixXd> MFF(mesh->cell(iT)->n_edges(), Eigen::MatrixXd::Zero(hho.nlocal_edge_dofs(), hho.nlocal_edge_dofs()));
371  std::vector<Eigen::MatrixXd> MTT_on_F(mesh->cell(iT)->n_edges(), Eigen::MatrixXd::Zero(hho.nlocal_cell_dofs(), hho.nlocal_cell_dofs()));
372  std::vector<Eigen::MatrixXd> MFT(mesh->cell(iT)->n_edges(), Eigen::MatrixXd::Zero(hho.nlocal_edge_dofs(), hho.nhighorder_dofs()));
373 
374  // Compute boundary terms
375  for (size_t ilF = 0; ilF < mesh->cell(iT)->n_edges(); ilF++) {
376  // Offset for face unknowns
377  const size_t offset_F = hho.nlocal_cell_dofs() + ilF * hho.nlocal_edge_dofs();
378  const size_t iF = mesh->cell(iT)->edge(ilF)->global_index();
379  const auto& nTF = mesh->cell(iT)->edge_normal(ilF);
380 
381  // Compute face quadrature points and values of basis functions (and gradients) at these points
382  std::vector<HybridCore::qrule> quadF = hho.edge_qrule(iF, 2*hho.K()+2);
383  size_t nbqF = quadF.size();
384 
385  std::vector<Eigen::VectorXd> phiT_quadF = hho.basis_quad('T', iT, quadF, hho.nhighorder_dofs());
386  std::vector<Eigen::VectorXd> phiF_quadF = hho.basis_quad('F', iF, quadF, hho.nlocal_edge_dofs());
387  std::vector<Eigen::MatrixXd> dphiT_quadF = hho.diff_grad_quad(iT, quadF, identity, hho.ngradient_dofs());
388 
389  // kappa*nTF on quadrature points
390  std::vector<Eigen::Vector2d> kappa_nTF_quadF(nbqF, Eigen::Vector2d::Zero());
391  for (size_t iqn = 0; iqn < nbqF; iqn++){
392  kappa_nTF_quadF[iqn] = kappa(quadF[iqn].x, quadF[iqn].y)*nTF;
393  }
394 
395  for (size_t i = 0; i < hho.ngradient_dofs(); i++) {
396  for (size_t j = 0; j < hho.nlocal_edge_dofs(); j++) {
397  for (size_t iqn=0; iqn < nbqF; iqn++){
398  BG(i, offset_F + j) += quadF[iqn].w * (dphiT_quadF[i].col(iqn)).dot(kappa_nTF_quadF[iqn]) * phiF_quadF[j](iqn);
399  }
400  }
401  for (size_t j = 0; j < hho.nlocal_cell_dofs(); j++) {
402  for (size_t iqn = 0; iqn < nbqF; iqn++){
403  BG(i,j) -= quadF[iqn].w * (dphiT_quadF[i].col(iqn)).dot(kappa_nTF_quadF[iqn]) * phiT_quadF[j](iqn);
404  }
405  }
406  }
407 
408  // Face mass matrix, and cell-face mass matrix, for stabilisation term below
409  MFF[ilF] = hho.gram_matrix(phiF_quadF, phiF_quadF, hho.nlocal_edge_dofs(), hho.nlocal_edge_dofs(), quadF, true);
410  MFT[ilF] = hho.gram_matrix(phiF_quadF, phiT_quadF, hho.nlocal_edge_dofs(), hho.nhighorder_dofs(), quadF, false);
411  MTT_on_F[ilF] = hho.gram_matrix(phiT_quadF, phiT_quadF, hho.nlocal_cell_dofs(), hho.nlocal_cell_dofs(), quadF, true);
412  }
413 
414  //------------------------------------------------------------------------------
415  // Solve for gradient reconstruction
416  Eigen::MatrixXd GT = MG.ldlt().solve(BG);
417  Eigen::MatrixXd ATF = BG.transpose() * GT;
418  //------------------------------------------------------------------------------
419 
420  //------------------------------------------------------------------------------
421  // Compute stabilization terms
422  Eigen::MatrixXd STF = Eigen::MatrixXd::Zero(local_dofs, local_dofs);
423 
424  // Cell residual delta_T^l = pi_T^l (rT uT) - u_T
425  Eigen::LDLT<Eigen::MatrixXd> piTL;
426  piTL.compute(MTT.topLeftCorner(hho.nlocal_cell_dofs(), hho.nlocal_cell_dofs()));
427  // start with pi_T^l (rT uT) then subtract u_T
428  // (note that rT calculated ici has zero average, not the average of u_T, but
429  // since it also appears in delta^k_TF, the incorrect averages cancel out)
430  Eigen::MatrixXd deltaTL = piTL.solve(MTT.block(0, 1, hho.nlocal_cell_dofs(), hho.ngradient_dofs()) * GT);
431  deltaTL.topLeftCorner(hho.nlocal_cell_dofs(), hho.nlocal_cell_dofs()) -=
432  Eigen::MatrixXd::Identity(hho.nlocal_cell_dofs(), hho.nlocal_cell_dofs());
433 
434  for (size_t ilF = 0; ilF < mesh->cell(iT)->n_edges(); ilF++) {
435  auto hF = mesh->cell(iT)->get_edge_measure(ilF);
436  auto xF = mesh->cell(iT)->get_edge_midpoint(ilF);
437  auto kappa_F = kappa(xF.x(), xF.y()).trace();
438 
439  // Face residual delta_TF^k = pi_F^k (rT uT) - u_F
440  Eigen::LDLT<Eigen::MatrixXd> piFK;
441  piFK.compute(MFF[ilF]);
442  // start with pi_F^k (rT uT)
443  Eigen::MatrixXd deltaTFK = piFK.solve(MFT[ilF].block(0, 1, hho.nlocal_edge_dofs(), hho.ngradient_dofs()) * GT);
444  deltaTFK.block(0, hho.nlocal_cell_dofs() + ilF * hho.nlocal_edge_dofs(), hho.nlocal_edge_dofs(), hho.nlocal_edge_dofs()) -=
445  Eigen::MatrixXd::Identity(hho.nlocal_edge_dofs(), hho.nlocal_edge_dofs());
446 
447  // The integral over F of (deltaTFK uT -deltaTL uT)*(deltaTFK vT - deltaTL vT) is decomposed into the integrals of
448  // deltaTFK uT * deltaTFK vT, -deltaTFK uT * deltaTL vT, -deltaTL uT * deltaTFK vT, deltaTL uT * deltaTL vT
449  // The two negative terms are transpose of each other
450  Eigen::MatrixXd cross_term = deltaTFK.transpose() * MFT[ilF].topLeftCorner(hho.nlocal_edge_dofs(), hho.nlocal_cell_dofs()) * deltaTL;
451  STF += (kappa_F / hF) *
452  ( deltaTFK.transpose() * MFF[ilF] * deltaTFK - cross_term - cross_term.transpose()
453  + deltaTL.transpose() * MTT_on_F[ilF] * deltaTL );

483 Eigen::VectorXd DiffusionEquation::load_operator(HybridCore &hho, const size_t iT) const {
484  // Load for the cell DOFs (first indices) and face DOFs (last indices)
485  const auto mesh = hho.get_mesh_ptr();
486  size_t cell_edge_dofs = hho.nlocal_cell_dofs() + mesh->cell(iT)->n_edges()*hho.nlocal_edge_dofs();
487  Eigen::VectorXd b = Eigen::VectorXd::Zero(cell_edge_dofs);
488 
489  // Quadrature points and values of cell basis functions at these points
490  std::vector<HybridCore::qrule> quadT = hho.cell_qrule(iT, 2*hho.K()+10);
491  size_t nbq = quadT.size();
492  std::vector<Eigen::VectorXd> phi_quadT = hho.basis_quad('T', iT, quadT, hho.nlocal_cell_dofs());
493 
494  // Value of source times quadrature weights at the quadrature points
495  Eigen::VectorXd weight_source_quad = Eigen::VectorXd::Zero(nbq);
496  for (size_t iqn = 0; iqn < nbq; iqn++){
497  weight_source_quad(iqn) = quadT[iqn].w * source(quadT[iqn].x, quadT[iqn].y);
498  }
499 
500  for (size_t i=0; i < hho.nlocal_cell_dofs(); i++){
501  b(i) = weight_source_quad.dot(phi_quadT[i]);
502  }
503  // Boundary values, if we have a boundary cell
504  if (mesh->cell(iT)->is_boundary()){
505  if (BC==0){
506  // Dirichlet BCs: no source terms on these edges
507  } else if (BC==1) {
508  // Neumann BCs
509  for (size_t ilF = 0; ilF < mesh->cell(iT)->n_edges(); ilF++) {
510  const size_t iF = mesh->cell(iT)->edge(ilF)->global_index();
511  // BC on boundary faces
512  if (mesh->cell(iT)->is_boundary_edge(ilF)){
513  // Offset for face unknowns
514  const size_t offset_F = hho.nlocal_cell_dofs() + ilF * hho.nlocal_edge_dofs();
515  // Normal to the face
516  const auto& nTF = mesh->cell(iT)->edge_normal(ilF);
517  // for each DOF of the boundary face
518  for (size_t i = 0; i < hho.nlocal_edge_dofs(); i++){
519  const auto& phi_i = hho.edge_basis(iF, i);
520  b(offset_F + i) = hho.integrate_over_edge(iF, [&](double x, double y){
521  return nTF.dot(kappa(x,y) * grad_exact_solution(x,y)) * phi_i(x,y);
522  });
523  }
524  }
525  }
526  }
527  }
528  for (size_t i=0; i< size_t(b.size()); i++){
529  if (std::isnan(b[i]))
530  std::cout << "The index "<< i << " of load_operator is nan"<<std::endl;
531  }
532 
533  return b;
534 }

536 double DiffusionEquation::H1Norm(HybridCore& hho, const Eigen::VectorXd XTF) {
537  const auto mesh = hho.get_mesh_ptr();
538  double value = 0.0;
539  for (size_t iT = 0; iT < mesh-> n_cells(); iT++) {
540 
541  // Identity tensor
542  tensor_function_type identity = [&](double x, double y) {
543  return Eigen::Matrix2d::Identity();
544  };
545 
546  // Compute cell quadrature points and values of cell basis functions and gradients
547  std::vector<HybridCore::qrule> quadT = hho.cell_qrule(iT, 2*hho.Ldeg());
548  std::vector<Eigen::MatrixXd> dphiT_quadT = hho.diff_grad_quad(iT, quadT, identity, hho.nlocal_cell_dofs()-1);
549 
550  // CELL CONTRIBUTION
551  //
552  // Stiffness matrix
553  Eigen::MatrixXd MG = hho.gram_matrix(dphiT_quadT, dphiT_quadT, hho.nlocal_cell_dofs()-1, hho.nlocal_cell_dofs()-1, quadT, true);
554 
555  // Cell indices
556  Eigen::VectorXd XTF_T = XTF.segment(iT*hho.nlocal_cell_dofs(),hho.nlocal_cell_dofs());
557  Eigen::VectorXd XTF_T_non_const = XTF_T.segment(1,hho.nlocal_cell_dofs()-1);
558 
559  value += XTF_T_non_const.dot(MG * XTF_T_non_const);
560 
561  // Faces contributions
562 
563  for (size_t ilF=0; ilF < mesh->cell(iT)->n_edges() ; ilF++){
564  size_t iF = mesh->cell(iT)->edge(ilF)->global_index();
565  auto hF = mesh->edge(iF)->measure();
566  // Face quadrature points and values of cell and face basis functions at these points
567  std::vector<HybridCore::qrule> quadF = hho.edge_qrule(iF, 2*hho.K()+2);
568  std::vector<Eigen::VectorXd> phiT_quadF = hho.basis_quad('T', iT, quadF, hho.nlocal_cell_dofs());
569  std::vector<Eigen::VectorXd> phiF_quadF = hho.basis_quad('F', iF, quadF, hho.nlocal_edge_dofs());
570 
571  // Face, cell and face-cell Gram matrix on F
572  Eigen::MatrixXd MFF = hho.gram_matrix(phiF_quadF, phiF_quadF, hho.nlocal_edge_dofs(), hho.nlocal_edge_dofs(), quadF, true);
573  Eigen::MatrixXd MFT = hho.gram_matrix(phiF_quadF, phiT_quadF, hho.nlocal_edge_dofs(), hho.nlocal_cell_dofs(), quadF, false);
574  Eigen::MatrixXd MTT_on_F = hho.gram_matrix(phiT_quadF, phiT_quadF, hho.nlocal_cell_dofs(), hho.nlocal_cell_dofs(), quadF, true);
575 
576  // Face indices
577  Eigen::VectorXd XTF_F = XTF.segment(hho.ntotal_cell_dofs() + iF * hho.nlocal_edge_dofs(), hho.nlocal_edge_dofs());
578 
579  double valF = XTF_T.dot(MTT_on_F * XTF_T) - 2 * XTF_F.dot(MFT * XTF_T) + XTF_F.dot(MFF * XTF_F);
580 
581  value += valF / hF;
582  }
583  }
584  return sqrt(value);
585 
586 }
