120 Eigen::VectorXd HHO_Diffusion::solve(HybridCore &hho) {
121 
122  boost::timer::cpu_timer timer; // Time the matrix assembly
123  boost::timer::cpu_timer timerint;
124  const auto mesh = hho.get_mesh_ptr();
125  aT.resize(mesh->n_cells());
126 
127  //--------------- PREPARE SYSTEM ------------------------//
128 
129  // System matrix
130  Eigen::SparseMatrix<double> GlobMat(hho.ntotal_edge_dofs(), hho.ntotal_edge_dofs());
131  std::vector<Eigen::Triplet<double>> triplets_GlobMat;
132  // If static condensation (L>=0): matrix to recover cell unknowns
133  // If barycentric elimination (L=-1): matrix to recover cell unknowns
134  Eigen::SparseMatrix<double> ScBeMat(hho.ntotal_cell_dofs(), hho.ntotal_edge_dofs());
135  std::vector<Eigen::Triplet<double>> triplets_ScBe;
136 
137  // Source terms for the system, and for recovering cell unknowns from static condensation
138  Eigen::VectorXd GlobRHS = Eigen::VectorXd::Zero(hho.ntotal_edge_dofs());
139  Eigen::VectorXd ScRHS = Eigen::VectorXd::Zero(hho.ntotal_cell_dofs());
140 
141  // Global quadrature rule for the cells
142  Eigen::VectorXd cell_quadrature = Eigen::VectorXd::Zero(hho.ntotal_cell_dofs());
143 
144 
145  //-------------- ASSEMBLE LOCAL CONTRIBUTIONS -------------//
146  auto total_measure = 0.0;
147 
148  for (size_t iT = 0; iT < mesh->n_cells(); iT++) {
149  Cell* iCell = mesh->cell(iT);
150 
151  total_measure += iCell->measure();
152 
153  // Total number of face degrees of freedom local to this cell (adjacent faces to the cell)
154  size_t nlocal_edges = iCell->n_edges();
155  size_t edge_dofs = nlocal_edges * hho.nlocal_edge_dofs();
156 
157  // Local bilinear form and source term
158  aT[iT] = diffusion_operator(hho, iT);
159  Eigen::VectorXd bT = load_operator(hho, iT);
160 
161  // Local matrix and right-hand side on the face unknowns
162  Eigen::MatrixXd MatF = Eigen::MatrixXd::Zero(edge_dofs,edge_dofs);
163  Eigen::VectorXd bF = Eigen::VectorXd::Zero(edge_dofs);
164 
165  if (hho.L()>=0) {
166  // STATIC CONDENSATION OF ELEMENT UNKNOWNS
167 
168  // Perform static condensation
169  Eigen::MatrixXd ATT = aT[iT].topLeftCorner(hho.nlocal_cell_dofs(), hho.nlocal_cell_dofs());
170  Eigen::MatrixXd ATF = aT[iT].topRightCorner(hho.nlocal_cell_dofs(), edge_dofs);
171  Eigen::MatrixXd AFF = aT[iT].bottomRightCorner(edge_dofs, edge_dofs);
172 
173  Eigen::PartialPivLU<Eigen::MatrixXd> invATT;
174  invATT.compute(ATT);
175 
176  Eigen::MatrixXd invATT_ATF = invATT.solve(ATF);
177  Eigen::VectorXd invATT_bTcell = invATT.solve(bT.head(hho.nlocal_cell_dofs()));
178  MatF = AFF - ATF.transpose() * invATT_ATF;
179 
180  bF = bT.tail(edge_dofs) - ATF.transpose() * invATT_bTcell;
181 
182  // Assemble static condensation operator
183  ScRHS.segment(iT * hho.nlocal_cell_dofs(), hho.nlocal_cell_dofs()) = invATT_bTcell;
184  for (size_t i = 0; i < hho.nlocal_cell_dofs(); i++) {
185  for (size_t jlF = 0; jlF < nlocal_edges; jlF++) {
186  const size_t jF = iCell->edge(jlF)->global_index();
187  for (size_t jk = 0; jk < hho.nlocal_edge_dofs(); jk++) {
188  const size_t jLocal = jlF * hho.nlocal_edge_dofs() + jk;
189  const size_t jGlobal = jF * hho.nlocal_edge_dofs() + jk;
190  triplets_ScBe.emplace_back(iT * hho.nlocal_cell_dofs() + i, jGlobal, invATT_ATF(i, jLocal));
191  }
192  }
193  }
194  } else {
195  // BARYCENTRIC ELIMINATION OF ELEMENT UNKNOWNS
196  // Create reduction matrix: 1+nlocal_edges * nlocal_edges matrix with the coefficients on the first row, and the identity below. When multiplied by the face unknowns, return cell and face unknowns
197  Eigen::MatrixXd red_matT = Eigen::MatrixXd::Zero(1+nlocal_edges,nlocal_edges);
198  red_matT.row(0) = hho.compute_weights(iT);
199  red_matT.bottomRightCorner(nlocal_edges,nlocal_edges) = Eigen::MatrixXd::Identity(nlocal_edges,nlocal_edges);
200 
201  bF = red_matT.transpose() * bT;
202  MatF = red_matT.transpose() * aT[iT] * red_matT;
203 
204  // To recover cell unknown
205  for (size_t jlF = 0; jlF < nlocal_edges; jlF++) {
206  const size_t jF = iCell->edge(jlF)->global_index();
207  const size_t jGlobal = jF * hho.nlocal_edge_dofs();
208  triplets_ScBe.emplace_back(iT, jGlobal, red_matT(0,jlF));
209  }
210 
211  }
212 
213  // Assemble local contribution into global matrix
214  for (size_t ilF = 0; ilF < nlocal_edges; ilF++) {
215  const size_t iF = iCell->edge(ilF)->global_index();
216  for (size_t ik = 0; ik < hho.nlocal_edge_dofs(); ik++) {
217  const size_t iLocal = ilF * hho.nlocal_edge_dofs() + ik;
218  const size_t iGlobal = iF * hho.nlocal_edge_dofs() + ik;
219  for (size_t jlF = 0; jlF < nlocal_edges; jlF++) {
220  const size_t jF = iCell->edge(jlF)->global_index();
221  for (size_t jk = 0; jk < hho.nlocal_edge_dofs(); jk++) {
222  const size_t jLocal = jlF * hho.nlocal_edge_dofs() + jk;
223  const size_t jGlobal = jF * hho.nlocal_edge_dofs() + jk;
224  triplets_GlobMat.emplace_back(iGlobal, jGlobal, MatF(iLocal, jLocal));
225  }
226  }
227  GlobRHS(iGlobal) += bF(iLocal);
228  }
229  }
230 
231  // Record cell quadrature
232  if (BC==1){
233  for (size_t i = 0; i < hho.nlocal_cell_dofs(); i++) {
234  const auto& phi_i = hho.cell_basis(iT, i);
235  cell_quadrature(iT * hho.nlocal_cell_dofs() + i) = hho.integrate_over_cell(iT, phi_i);
236  }
237  }
238 
239  }
240 
241  if (BC==1){
242  // Neumann BC: remove a row in the matrix and fix the first degree of freedom
243  triplets_GlobMat.erase(std::remove_if(std::begin(triplets_GlobMat), std::end(triplets_GlobMat),
244  [](const auto& x) { return (x.row() == 0); }), std::end(triplets_GlobMat));
245  triplets_GlobMat.emplace_back(0, 0, 1);
246  GlobRHS(0) = 0;
247  }
248 
249  // Assemble the global linear system (without BC), and matrix to recover statically-condensed cell dofs
250  GlobMat.setFromTriplets(std::begin(triplets_GlobMat), std::end(triplets_GlobMat));
251  ScBeMat.setFromTriplets(std::begin(triplets_ScBe), std::end(triplets_ScBe));
252 
253  //-------------- TREATMENT OF BOUNDARY CONDITIONS -------------//
254 
255  // If Dirichlet, the final system is only posed on the interior edge unknowns and we have to subtract from the source
256  // term the contribution of the boundary values
257  // If Neumann, the final system is posed on all edge unknowns
258 
259  size_t nb_unknowns = 0;
260  size_t nb_fixed_dofs = 0;
261  Eigen::SparseMatrix<double> A;
262  Eigen::VectorXd B;
263  Eigen::VectorXd UDir;
264 
265  if (BC==0){
266  // Dirichlet boundary conditions
267  nb_unknowns = hho.ninternal_edge_dofs();
268  nb_fixed_dofs = hho.nboundary_edge_dofs();
269  A = GlobMat.topLeftCorner(nb_unknowns, nb_unknowns);
270 
271  // Boundary value: UDir corresponds to the L2 projection of the exact solution on the polynomial spaces on the edges
272  UDir = Eigen::VectorXd::Zero(nb_fixed_dofs);
273  std::vector<Edge *> b_edges = mesh->get_b_edges(); // List of boundary edges
274 
275  for (size_t ibF = 0; ibF < mesh->n_b_edges(); ibF++){
276 
277  size_t iF = b_edges[ibF]->global_index();
278  // Mass matrix and boundary values
279  std::vector<HybridCore::qrule> quadF = hho.edge_qrule(iF, 2*hho.K());
280  std::vector<Eigen::ArrayXd> phiF_quadF = hho.basis_quad('F', iF, quadF, hho.nlocal_edge_dofs());
281  Eigen::MatrixXd MFF = hho.gram_matrix(phiF_quadF, phiF_quadF, hho.nlocal_edge_dofs(), hho.nlocal_edge_dofs(), quadF, true);
282 
283  // Compute (uexact, phi_i)_F for all edge basis functions phi_i
284  Eigen::VectorXd RHS_UDirF = Eigen::VectorXd::Zero(hho.nlocal_edge_dofs());
285  for (size_t j = 0; j < hho.nlocal_edge_dofs(); j++){
286  const auto& phi_j = hho.edge_basis(iF, j);
287  RHS_UDirF(j) = hho.integrate_over_edge(iF, [this, &phi_j](double x, double y) {
288  return phi_j(x,y) * exact_solution(x,y); });
289  }
290  // Project exact solution
291  UDir.segment(ibF * hho.nlocal_edge_dofs(), hho.nlocal_edge_dofs()) = MFF.ldlt().solve(RHS_UDirF);
292 
293  }
294 
295  B = GlobRHS.segment(0, nb_unknowns) - GlobMat.topRightCorner(nb_unknowns, nb_fixed_dofs) * UDir;
296 
297  } else if (BC==1){
298  // We will solve the complete system
299  nb_unknowns = hho.ntotal_edge_dofs();
300  A = GlobMat;
301  B = GlobRHS;
302  UDir = Eigen::VectorXd::Zero(nb_fixed_dofs);
303  }
304 
305  _assembly_time = timer.elapsed().user + timer.elapsed().system;
306 
307  //-------------- SOLVE CONDENSED SYSTEM -------------//
308  timer.start();
309 
310  Eigen::VectorXd xF = Eigen::VectorXd::Zero(nb_unknowns);
311 
312 // if (solver_type == "ma41") {
313 // Eigen::MA41<Eigen::SparseMatrix<double>, Eigen::VectorXd> solver;
314 // solver.analyzePattern(A);
315 // solver.factorize(A);
316 // xF = solver.solve(B);
317 // } else {
318  Eigen::BiCGSTAB<Eigen::SparseMatrix<double> > solver;
319  solver.compute(A);
320  xF = solver.solve(B);
321  std::cout << "#iterations: " << solver.iterations() << std::endl;
322  std::cout << "estimated error: " << solver.error() << std::endl;
323 // }
324  _solving_error = (A * xF - B).norm();
325  // Recover the fixed boundary values, cell unknowns (from static condensation/barycentric elimination)
326  Eigen::VectorXd Xh = Eigen::VectorXd::Zero(hho.ntotal_dofs());
327  Xh.tail(nb_fixed_dofs) = UDir;
328  Xh.segment(hho.ntotal_cell_dofs(), nb_unknowns) = xF;
329  if (hho.L()>=0) {
330  Xh.head(hho.ntotal_cell_dofs()) = ScRHS - ScBeMat * Xh.tail(hho.ntotal_edge_dofs());
331  } else {
332  Xh.head(hho.ntotal_cell_dofs()) = ScBeMat * Xh.tail(hho.ntotal_edge_dofs());
333  }
334 
335  // Only Neumann: translate to get the proper average
336  if (BC==1){
337  // Compute average to translate
338  auto average = cell_quadrature.dot(Xh.head(hho.ntotal_cell_dofs())) / total_measure;
339  auto average_exact_sol = hho.integrate_over_domain([&](auto x, auto y) {
340  return exact_solution(x,y);
341  }) / total_measure;
342 
343  // Translate the cells
344  for (size_t iT = 0; iT < mesh->n_cells(); iT++) {
345  Xh(iT * hho.nlocal_cell_dofs()) += average_exact_sol - average;
346  }
347 
348  for (size_t iF = 0; iF < mesh->n_edges(); iF++) {
349  Xh(hho.ntotal_cell_dofs() + iF * hho.nlocal_edge_dofs()) += average_exact_sol - average;
350  }
351  }
352 
353  _solving_time = timer.elapsed().user + timer.elapsed().system; // Record the final solving time
354 
355 
356  return Xh;
357 }

363 Eigen::MatrixXd HHO_Diffusion::diffusion_operator(HybridCore &hho, const size_t iT) const {
364 
365  boost::timer::cpu_timer timeint;
366 
367  const auto mesh = hho.get_mesh_ptr();
368  Cell* cell = mesh->cell(iT);
369  const size_t nedgesT = cell->n_edges();
370 
371  // Total number of degrees of freedom local to this cell (cell and its adjacent faces)
372  size_t local_dofs = hho.nlocal_cell_dofs() + nedgesT * hho.nlocal_edge_dofs();
373 
374  //------------------- Initialisatons ------------------------------//
375 
376 //timeint.restart();
377 
378  // Diffusion in the cell.
379  std::function<Eigen::Matrix2d(double,double)> kappaT = [&](double x, double y){
380  // Constant in the cell
381  Eigen::Vector2d cellcenter = cell->center_mass();
382  return kappa(cellcenter.x(), cellcenter.y(), cell);
383  // Variable in the cell - the scheme may not provide optimal convergence rate if the diffusion is actually variable in the cell!
384  // If the diffusion is piecewise constant, choosing the previous version ensures that, for edge integrals, it's the value inside the cell that is computed (essential in case kappa is discontinuous across the edges)
385 // return kappa(x, y, cell);
386  };
387 
388  // Compute cell quadrature nodes and values of cell basis functions and gradients at these nodes
389  std::vector<HybridCore::qrule> quadT = hho.cell_qrule(iT, hho.Ldeg()+hho.K()+1);
390  std::vector<Eigen::ArrayXd> phi_quadT = hho.basis_quad('T', iT, quadT, hho.nhighorder_dofs());
391  std::vector<Eigen::ArrayXXd> dphiT_quadT = hho.grad_basis_quad(iT, quadT, hho.nhighorder_dofs());
392  // Diffusion tensor at the quadrature nodes
393  std::vector<Eigen::Matrix2d> kappaT_quadT(quadT.size());
394  std::transform(quadT.begin(), quadT.end(), kappaT_quadT.begin(),
395  [&kappaT,&cell](HybridCore::qrule qr) -> Eigen::MatrixXd { return kappaT(qr.x, qr.y); });
396 
397 
398 //itime[0] += timeint.elapsed();
399 //timeint.restart();
400 
401  // Preparing matrices to store face mass matrices (for face and cell basis functions), and face-cell mass matrices
402  // (will be useful for the stabilisation term)
403  std::vector<Eigen::MatrixXd> MFF(mesh->cell(iT)->n_edges(), Eigen::MatrixXd::Zero(hho.nlocal_edge_dofs(), hho.nlocal_edge_dofs()));
404  std::vector<Eigen::MatrixXd> MFT(mesh->cell(iT)->n_edges(), Eigen::MatrixXd::Zero(hho.nlocal_edge_dofs(), hho.nhighorder_dofs()));
405 
406 //itime[1] += timeint.elapsed();
407 //timeint.restart();
408 
409 
410  //-------------------- Compute PT, matrix of potential reconstruction ---------//
411 
412  // Stiffness matrix (kappaT dphi_i,dphi_j)_T for phi_i, phi_j up to degree K+1
413  Eigen::MatrixXd StiffT = hho.gram_matrix(dphiT_quadT, dphiT_quadT, hho.nhighorder_dofs(), hho.nhighorder_dofs(), quadT, true, kappaT_quadT);
414 
415  // Mass matrix of (phi_i,phi_j)_T for phi_i up to degree L and phi_j up to degree K+1
416  Eigen::MatrixXd MTT = hho.gram_matrix(phi_quadT, phi_quadT, hho.nlocal_cell_dofs(), hho.nhighorder_dofs(), quadT, true);
417 
418  // Row vector of (phi_j,1)_T for phi_j up to degree K+1, and LT^t*LT (used later to impose average condition on PT)
419  Eigen::VectorXd LT = (MTT.row(0)).transpose();
420  Eigen::MatrixXd LTtLT = LT * (LT.transpose());
421 
422  // Right-hand side: we start with volumetric terms (dphi_i,dphi_j)_T for phi_i up to degree K+1 and phi_j up to degree L
423  Eigen::MatrixXd BP = Eigen::MatrixXd::Zero(hho.nhighorder_dofs(), local_dofs);
424  BP.topLeftCorner(hho.nhighorder_dofs(), hho.nlocal_cell_dofs()) = StiffT.topLeftCorner(hho.nhighorder_dofs(), hho.nlocal_cell_dofs());
425 
426  // Boundary terms in BP
427  for (size_t ilF = 0; ilF < mesh->cell(iT)->n_edges(); ilF++) {
428  // Offset for face unknowns
429  const size_t offset_F = hho.nlocal_cell_dofs() + ilF * hho.nlocal_edge_dofs();
430  const size_t iF = mesh->cell(iT)->edge(ilF)->global_index();
431  const auto& nTF = mesh->cell(iT)->edge_normal(ilF);
432 
433  // Compute face quadrature nodes and values of basis functions (and gradients) at these nodes
434  std::vector<HybridCore::qrule> quadF = hho.edge_qrule(iF, 2*hho.K()+2);
435  size_t nbqF = quadF.size();
436 
437  std::vector<Eigen::ArrayXd> phiT_quadF = hho.basis_quad('T', iT, quadF, hho.nhighorder_dofs());
438  std::vector<Eigen::ArrayXd> phiF_quadF = hho.basis_quad('F', iF, quadF, hho.nlocal_edge_dofs());
439  std::vector<Eigen::ArrayXXd> dphiT_quadF = hho.grad_basis_quad(iT, quadF, hho.nhighorder_dofs());
440 
441  // kappaT*nTF on quadrature nodes
442  std::vector<Eigen::Vector2d> kappaT_nTF_quadF(nbqF, Eigen::Vector2d::Zero());
443  for (size_t iqn = 0; iqn < nbqF; iqn++){
444  kappaT_nTF_quadF[iqn] = kappaT(quadF[iqn].x, quadF[iqn].y)*nTF;
445  }
446 
447  for (size_t i = 1; i < hho.nhighorder_dofs(); i++) {
448  // We do not need i=0 because it corresponds to dphi_i=0
449  for (size_t j = 0; j < hho.nlocal_cell_dofs(); j++) {
450  for (size_t iqn = 0; iqn < nbqF; iqn++){
451  BP(i,j) -= quadF[iqn].w * (dphiT_quadF[i].col(iqn).matrix()).dot(kappaT_nTF_quadF[iqn]) * phiT_quadF[j](iqn);
452  }
453  }
454  for (size_t j = 0; j < hho.nlocal_edge_dofs(); j++) {
455  for (size_t iqn=0; iqn < nbqF; iqn++){
456  BP(i, offset_F + j) += quadF[iqn].w * (dphiT_quadF[i].col(iqn).matrix()).dot(kappaT_nTF_quadF[iqn]) * phiF_quadF[j](iqn);
457  }
458  }
459  }
460 
461  // Face mass matrix, and cell-face mass matrix, for stabilisation term below
462  MFF[ilF] = hho.gram_matrix(phiF_quadF, phiF_quadF, hho.nlocal_edge_dofs(), hho.nlocal_edge_dofs(), quadF, true);
463  MFT[ilF] = hho.gram_matrix(phiF_quadF, phiT_quadF, hho.nlocal_edge_dofs(), hho.nhighorder_dofs(), quadF, false);
464  }
465 
466  // Compute PT
467  double scalT = StiffT.trace() / std::pow(LT.norm(), 2);
468  BP.topLeftCorner(hho.nhighorder_dofs(), hho.nlocal_cell_dofs()) +=
469  scalT * LTtLT.topLeftCorner(hho.nhighorder_dofs(), hho.nlocal_cell_dofs());
470  Eigen::MatrixXd PT = ((StiffT+scalT*LTtLT).ldlt()).solve(BP);
471 
472  // Consistent component (K \nabla pT, \nabla pT)_T in local bilinear form
473  Eigen::MatrixXd ATF = PT.transpose() * StiffT * PT;
474 
475  //-------------------- Compute stabilisation term sT ---------//
476 
477  Eigen::MatrixXd STF = Eigen::MatrixXd::Zero(local_dofs, local_dofs);
478 
479  // Cell residual delta_T^l = pi_T^l (rT uT) - u_T
480  Eigen::MatrixXd MTTKinv = ( MTT.topLeftCorner(hho.nlocal_cell_dofs(), hho.nlocal_cell_dofs()) ).inverse();
481  Eigen::MatrixXd deltaTL = MTTKinv * MTT * PT;
482  deltaTL.topLeftCorner(hho.nlocal_cell_dofs(), hho.nlocal_cell_dofs()) -=
483  Eigen::MatrixXd::Identity(hho.nlocal_cell_dofs(), hho.nlocal_cell_dofs());
484 
485  for (size_t ilF = 0; ilF < nedgesT; ilF++) {
486  auto hF = mesh->cell(iT)->edge(ilF)->measure();
487  auto xF = mesh->cell(iT)->edge(ilF)->center_mass();
488 
489  auto kappa_TF = kappaT(xF.x(), xF.y()).trace();
490 
491  // Face residual delta_TF^k = pi_F^k (rT uT) - u_F
492  Eigen::MatrixXd MFFinv = MFF[ilF].inverse();
493  Eigen::MatrixXd deltaTFK = MFFinv * MFT[ilF] * PT;
494  deltaTFK.block(0, hho.nlocal_cell_dofs() + ilF * hho.nlocal_edge_dofs(), hho.nlocal_edge_dofs(), hho.nlocal_edge_dofs()) -=
495  Eigen::MatrixXd::Identity(hho.nlocal_edge_dofs(), hho.nlocal_edge_dofs());
496 
497  // Stabilisation term
498  Eigen::MatrixXd dTFKminusdTL = deltaTFK - MFFinv * MFT[ilF].topLeftCorner(hho.nlocal_edge_dofs(), hho.nlocal_cell_dofs()) * deltaTL;
499 
500  STF += (kappa_TF / hF) * dTFKminusdTL.transpose() * MFF[ilF] * dTFKminusdTL;
501 
502  }
503 //itime[2] += timeint.elapsed();
504 //timeint.restart();
505 
506  // Adjust local bilinear form with stabilisation term
507  ATF += STF;
508 
509  return ATF;

518 Eigen::VectorXd HHO_Diffusion::load_operator(HybridCore &hho, const size_t iT) const {
519  // Load for the cell DOFs (first indices) and face DOFs (last indices)
520  const auto mesh = hho.get_mesh_ptr();
521  Cell* cell = mesh->cell(iT);
522  size_t cell_edge_dofs = hho.nlocal_cell_dofs() + cell->n_edges()*hho.nlocal_edge_dofs();
523  Eigen::VectorXd b = Eigen::VectorXd::Zero(cell_edge_dofs);
524 
525  // Quadrature nodes and values of cell basis functions at these nodes
526  std::vector<HybridCore::qrule> quadT = hho.cell_qrule(iT, 2*hho.K()+10);
527  size_t nbq = quadT.size();
528  std::vector<Eigen::ArrayXd> phiT_quadT = hho.basis_quad('T', iT, quadT, hho.nlocal_cell_dofs());
529 
530  // Value of source times quadrature weights at the quadrature nodes
531  Eigen::VectorXd weight_source_quad = Eigen::VectorXd::Zero(nbq);
532  for (size_t iqn = 0; iqn < nbq; iqn++){
533  weight_source_quad(iqn) = quadT[iqn].w * source(quadT[iqn].x, quadT[iqn].y, cell);
534  }
535 
536  for (size_t i=0; i < hho.nlocal_cell_dofs(); i++){
537  b(i) = weight_source_quad.dot(phiT_quadT[i].matrix());
538  }
539  // Boundary values, if we have a boundary cell
540  if (cell->is_boundary()){
541  if (BC==0){
542  // Dirichlet BCs: no source terms on these edges
543  } else if (BC==1) {
544  // Neumann BCs
545  for (size_t ilF = 0; ilF < cell->n_edges(); ilF++) {
546  const size_t iF = cell->edge(ilF)->global_index();
547  // BC on boundary faces
548  if (cell->edge(ilF)->is_boundary()){
549  // Offset for face unknowns
550  const size_t offset_F = hho.nlocal_cell_dofs() + ilF * hho.nlocal_edge_dofs();
551  // Normal to the face
552  const auto& nTF = cell->edge_normal(ilF);
553  // for each DOF of the boundary face
554  for (size_t i = 0; i < hho.nlocal_edge_dofs(); i++){
555  const auto& phi_i = hho.edge_basis(iF, i);
556  b(offset_F + i) = hho.integrate_over_edge(iF, [&](double x, double y){
557  return nTF.dot(kappa(x,y,cell) * grad_exact_solution(x,y,cell)) * phi_i(x,y);
558  });
559  }
560  }
561  }
562  }
563  }
564 
565  return b;
566 }

582 Eigen::VectorXd HHO_Diffusion::VertexValues(HybridCore& hho, const Eigen::VectorXd Xh, const char from_dofs) {
583  const auto mesh = hho.get_mesh_ptr();
584  Eigen::VectorXd function_vertex = Eigen::VectorXd::Zero(mesh->n_vertices());
585 
586  if (from_dofs == 'T'){
587  // compute from the cell polynomials, averaging all values coming from the cells around each vertex
588  for (size_t iV = 0; iV < mesh->n_vertices(); iV++){
589  auto xV = mesh->vertex(iV)->coords();
590  auto cList = mesh->vertex(iV)->get_cells();
591  auto weight = cList.size();
592  for (size_t ilT = 0; ilT < weight; ilT++){
593  auto temp = hho.evaluate_in_cell(Xh, cList[ilT]->global_index(), xV.x(), xV.y());
594  function_vertex(iV) += temp;
595  }
596  function_vertex(iV) = function_vertex(iV)/(weight);
597  }
598 
599  }else{
600  // compute from the edge polynomials, averaging all values coming from the edges around each vertex
601  for (size_t iV = 0; iV < mesh->n_vertices(); iV++){
602  auto xV = mesh->vertex(iV)->coords();
603  auto eList = mesh->vertex(iV)->get_edges();
604  double weight = eList.size();
605  for (size_t ilF = 0; ilF < weight; ilF++){
606  auto temp = hho.evaluate_in_edge(Xh, eList[ilF]->global_index(), xV.x(), xV.y());
607  function_vertex(iV) += temp;
608  }
609  function_vertex(iV) = function_vertex(iV)/(weight);
610  }
611  }
612 
613  return function_vertex;
