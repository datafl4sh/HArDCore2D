1 // Core data structures and methods required to implement hybrid schemes in 2D (polynomial unknowns
2 // in the cells and on the edges, such as Hybrid High-order (HHO) schemes).
3 //
4 // Provides:
5 // - Hybrid polynomial basis functions (on the cells and faces of the mesh)
6 // - Generic routines to create quadrature points over cells and faces of the mesh
7 // - Interpolation of general functions onto the HHO space
8 // - Methods for integrating, evaluating, and computing norms of HHO solutions
9 //
10 // Author: Jerome Droniou (jerome.droniou@monash.edu)

259 void HybridCore::quadrature_over_cell(const size_t iT, const Function& f) const {
260  assert(iT < _mesh_ptr->n_cells());
261  std::vector<HybridCore::qrule> quadT = cell_qrule(iT, 2 * _Ldeg + 5);
262  for (size_t iqn = 0; iqn < quadT.size(); iqn++) {
263  f(iqn, quadT[iqn].x, quadT[iqn].y, quadT[iqn].w);
264  }
265 }

311 template<typename Function>
312 Eigen::VectorXd HybridCore::interpolate(const Function& f, size_t doe) const {
313  Eigen::VectorXd XTF = Eigen::VectorXd::Zero(_ntotal_dofs);
314 
315  for (size_t iT = 0; iT < _mesh_ptr->n_cells(); iT++) {
316 
317  // Mass matrix in cell
318  std::vector<HybridCore::qrule> quadT = cell_qrule(iT, doe);
319  std::vector<Eigen::VectorXd> phi_quadT = basis_quad('T', iT, quadT, _nlocal_cell_dofs);
320  Eigen::MatrixXd MT = gram_matrix(phi_quadT, phi_quadT, _nlocal_cell_dofs, _nlocal_cell_dofs, quadT, true);
321 
322  // Vector of integrals of f against basis functions
323  Eigen::VectorXd bT = Eigen::VectorXd::Zero(_nlocal_cell_dofs);
324  for (size_t i = 0; i < _nlocal_cell_dofs; i++) {
325  const auto& phi_i = cell_basis(iT, i);
326  bT(i) += integrate_over_cell(iT, [&phi_i, &f](double x, double y) {
327  return phi_i(x,y) * f(x,y);
328  });
329  }
330 
331  // Vector of coefficients (on cell basis functions) of the L2(T) projection of f
332  Eigen::VectorXd UT = MT.ldlt().solve(bT);
333 
334  // Fill in the complete vector of DOFs
335  size_t offset_T = iT * _nlocal_cell_dofs;
336  XTF.segment(offset_T, _nlocal_cell_dofs) = UT;
337 
338  for (size_t ilF = 0; ilF < _mesh_ptr->cell(iT)->n_edges(); ilF++) {
339  size_t iF = _mesh_ptr->cell(iT)->edge(ilF)->global_index();
340 
341  // Mass matrix on face
342  std::vector<HybridCore::qrule> quadF = edge_qrule(iF, doe);
343  std::vector<Eigen::VectorXd> phiF_quadF = basis_quad('F', iF, quadF, _nlocal_edge_dofs);
344  Eigen::MatrixXd MF = gram_matrix(phiF_quadF, phiF_quadF, _nlocal_edge_dofs, _nlocal_edge_dofs, quadF, true);
345 
346  // Vector of integral of f against face basis functions
347  Eigen::VectorXd bF = Eigen::VectorXd::Zero(_nlocal_edge_dofs);
348  for (size_t i = 0; i < _nlocal_edge_dofs; i++) {
349  const auto& phi_i = edge_basis(iF, i);
350  bF(i) += integrate_over_edge(iF, [&phi_i, &f](double x, double y) {
351  return phi_i(x,y) * f(x,y);
352  });
353  }
354 
355  // Vector of coefficients (on cell basis functions) of the L2(F) projection of f
356  Eigen::VectorXd UF = MF.ldlt().solve(bF);
357 
358  // Fill in he complete vector of DOFs
359  XTF.segment(_ntotal_cell_dofs + iF * _nlocal_edge_dofs, _nlocal_edge_dofs) = UF;
360  }
361 
362  // Special case of L=-1, we replace the previously computed cell value with the proper average of face values
363  if ( _L==-1 ){
364  Eigen::VectorXd barycoefT = compute_weights(iT);
365  XTF(iT) = 0;
366  for (size_t ilF = 0; ilF < _mesh_ptr->cell(iT)->n_edges(); ilF++) {
367  size_t iF = _mesh_ptr->cell(iT)->edge(ilF)->global_index();
368  XTF(iT) += barycoefT(ilF) * XTF(_ntotal_cell_dofs + iF);
369  }
370  }
371 
372  }
373 
374  return XTF;
375 }
