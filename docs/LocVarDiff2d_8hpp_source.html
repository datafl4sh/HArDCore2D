119 Eigen::VectorXd LocVarDiff::solve(HybridCore &hho) {
120 
121  boost::timer timer; // Time the matrix assembly
122  const auto mesh = hho.get_mesh_ptr();
123 
124  // System matrix
125  Eigen::SparseMatrix<double> A(hho.ntotal_edge_dofs(), hho.ntotal_edge_dofs());
126  // If static condensation (L>=0): matrix to recover cell unknowns
127  // If barycentric elimination (L=-1): matrix to recover cell unknowns
128  Eigen::SparseMatrix<double> A_sc_be(hho.ntotal_cell_dofs(), hho.ntotal_edge_dofs());
129 
130  Eigen::VectorXd B = Eigen::VectorXd::Zero(hho.ntotal_edge_dofs());
131  Eigen::VectorXd B_sc = Eigen::VectorXd::Zero(hho.ntotal_cell_dofs());
132 
133  //------------------------------------------------------------------------------
134  // Assemble local contributions
135 
136  // Global quadrature rule for the cells
137 
138  Eigen::VectorXd cell_quadrature = Eigen::VectorXd::Zero(hho.ntotal_cell_dofs());
139 
140 
141  std::vector<Eigen::Triplet<double>> triplets;
142  std::vector<Eigen::Triplet<double>> triplets_sc_be; // Triplets to create A_sc or A_be
143 
144  auto total_measure = 0.0;
145 
146  for (size_t iT = 0; iT < mesh->n_cells(); iT++) {
147  Cell2D* iCell = mesh->cell(iT);
148 
149  total_measure += iCell->measure();
150 
151  // Total number of face degrees of freedom local to this cell (adjacent faces to the cell)
152  size_t nlocal_edges = iCell->n_edges();
153  size_t edge_dofs = nlocal_edges * hho.nlocal_edge_dofs();
154  // Compute local operators
155 
156  Eigen::VectorXd bT = load_operator(hho, iT);
157  Eigen::MatrixXd aT = diffusion_operator(hho, iT);
158 
159  // Local matrix and right-hand side on the face unknowns
160  Eigen::MatrixXd MatF = Eigen::MatrixXd::Zero(edge_dofs,edge_dofs);
161  Eigen::VectorXd bF = Eigen::VectorXd::Zero(edge_dofs);
162 
163  if (hho.L()>=0) {
164  // STATIC CONDENSATION OF ELEMENT UNKNOWNS
165 
166  // Perform static condensation
167  Eigen::MatrixXd ATT = aT.topLeftCorner(hho.nlocal_cell_dofs(), hho.nlocal_cell_dofs());
168  Eigen::MatrixXd ATF = aT.topRightCorner(hho.nlocal_cell_dofs(), edge_dofs);
169  Eigen::MatrixXd AFF = aT.bottomRightCorner(edge_dofs, edge_dofs);
170 
171  Eigen::PartialPivLU<Eigen::MatrixXd> invATT;
172  invATT.compute(ATT);
173 
174  Eigen::MatrixXd invATT_ATF = invATT.solve(ATF);
175  Eigen::VectorXd invATT_bTcell = invATT.solve(bT.head(hho.nlocal_cell_dofs()));
176  MatF = AFF - ATF.transpose() * invATT_ATF;
177 
178  bF = bT.tail(edge_dofs) - ATF.transpose() * invATT_bTcell;
179 
180  // Assemble static condensation operator
181  B_sc.segment(iT * hho.nlocal_cell_dofs(), hho.nlocal_cell_dofs()) = invATT_bTcell;
182  for (size_t i = 0; i < hho.nlocal_cell_dofs(); i++) {
183  for (size_t jlF = 0; jlF < nlocal_edges; jlF++) {
184  const size_t jF = iCell->edge(jlF)->global_index();
185  for (size_t jk = 0; jk < hho.nlocal_edge_dofs(); jk++) {
186  const size_t jLocal = jlF * hho.nlocal_edge_dofs() + jk;
187  const size_t jGlobal = jF * hho.nlocal_edge_dofs() + jk;
188  triplets_sc_be.emplace_back(iT * hho.nlocal_cell_dofs() + i, jGlobal, invATT_ATF(i, jLocal));
189  }
190  }
191  }
192  } else {
193  // BARYCENTRIC ELIMINATION OF ELEMENT UNKNOWNS
194  // Create reduction matrix: 1+nlocal_edges * nlocal_edges matrix with the coefficients on the first row, and the identity below. When multiplied by the face unknowns, return cell and face unknowns
195  Eigen::MatrixXd red_matT = Eigen::MatrixXd::Zero(1+nlocal_edges,nlocal_edges);
196  red_matT.row(0) = hho.compute_weights(iT);
197  red_matT.bottomRightCorner(nlocal_edges,nlocal_edges) = Eigen::MatrixXd::Identity(nlocal_edges,nlocal_edges);
198 
199  bF = red_matT.transpose() * bT;
200  MatF = red_matT.transpose() * aT * red_matT;
201 
202  // To recover cell unknown
203  for (size_t jlF = 0; jlF < nlocal_edges; jlF++) {
204  const size_t jF = iCell->edge(jlF)->global_index();
205  const size_t jGlobal = jF * hho.nlocal_edge_dofs();
206  triplets_sc_be.emplace_back(iT, jGlobal, red_matT(0,jlF));
207  }
208 
209  }
210 
211 
212  // Assemble local contribution into global matrix
213  for (size_t ilF = 0; ilF < nlocal_edges; ilF++) {
214  const size_t iF = iCell->edge(ilF)->global_index();
215  for (size_t ik = 0; ik < hho.nlocal_edge_dofs(); ik++) {
216  const size_t iLocal = ilF * hho.nlocal_edge_dofs() + ik;
217  const size_t iGlobal = iF * hho.nlocal_edge_dofs() + ik;
218  for (size_t jlF = 0; jlF < nlocal_edges; jlF++) {
219  const size_t jF = iCell->edge(jlF)->global_index();
220  for (size_t jk = 0; jk < hho.nlocal_edge_dofs(); jk++) {
221  const size_t jLocal = jlF * hho.nlocal_edge_dofs() + jk;
222  const size_t jGlobal = jF * hho.nlocal_edge_dofs() + jk;
223  triplets.emplace_back(iGlobal, jGlobal, MatF(iLocal, jLocal));
224  }
225  }
226  B(iGlobal) += bF(iLocal);
227  }
228  }
229 
230  // Record cell quadrature
231  if (BC==1){
232  for (size_t i = 0; i < hho.nlocal_cell_dofs(); i++) {
233  const auto& phi_i = hho.cell_basis(iT, i);
234  cell_quadrature(iT * hho.nlocal_cell_dofs() + i) = hho.integrate_over_cell(iT, phi_i);
235  }
236  }
237 
238  }
239  if (BC==0){
240  // Dirichlet boundary conditions: the equations for boundary faces are replaced
241  // with MF UF = (int of phi_i u)_i. This forces UF to represent the coefficients
242  // of the L2(F) projection of u on the basis (phi_i)_i
243  std::vector<Edge2D *> b_edges = mesh->get_b_edges(); // List of boundary edges
244  for (size_t ibF = 0; ibF < mesh->n_b_edges(); ibF++){
245  size_t iF = b_edges[ibF]->global_index();
246  // Mass matrix and boundary values
247  std::vector<HybridCore::qrule> quadF = hho.edge_qrule(iF, 2*hho.K());
248  std::vector<Eigen::VectorXd> phiF_quadF = hho.basis_quad('F', iF, quadF, hho.nlocal_edge_dofs());
249  Eigen::MatrixXd MF = hho.gram_matrix(phiF_quadF, phiF_quadF, hho.nlocal_edge_dofs(), hho.nlocal_edge_dofs(), quadF, true);
250  for (size_t j = 0; j < hho.nlocal_edge_dofs(); j++){
251  // Erase row in matrix and replace
252  size_t jGlobal = iF * hho.nlocal_edge_dofs() + j;
253  triplets.erase(std::remove_if(std::begin(triplets), std::end(triplets),
254  [&jGlobal](const auto& x) { return (x.row() == (int) jGlobal); }), std::end(triplets));
255  for (size_t k = 0; k < hho.nlocal_edge_dofs(); k++){
256  size_t kGlobal = iF * hho.nlocal_edge_dofs() + k;
257  triplets.emplace_back(jGlobal, kGlobal, MF(j,k));
258  }
259  // Replace boundary value
260  const auto& phi_j = hho.edge_basis(iF, j);
261  B(jGlobal) = hho.integrate_over_edge(iF, [this, &phi_j](double x, double y) {
262  return phi_j(x,y) * exact_solution(x,y); });
263  }
264 
265  }
266  } else if (BC==1){
267  // Remove a row in the matrix and fix the first degree of freedom
268  triplets.erase(std::remove_if(std::begin(triplets), std::end(triplets),
269  [](const auto& x) { return (x.row() == 0); }), std::end(triplets));
270  triplets.emplace_back(0, 0, 1);
271  B(0) = 0;
272  }
273  // Assemble the linear system
274  A.setFromTriplets(std::begin(triplets), std::end(triplets));
275  A_sc_be.setFromTriplets(std::begin(triplets_sc_be), std::end(triplets_sc_be));
276 
277  assembly_time = timer.elapsed(); // Record final assembly time
278  timer.restart(); // Time the solving
279 
280  // Solve the condensed system
281  Eigen::VectorXd xF = Eigen::VectorXd::Zero(hho.ntotal_edge_dofs());
282 // if (solver_type == "ma41") {
283 // Eigen::MA41<Eigen::SparseMatrix<double>, Eigen::VectorXd> solver;
284 // solver.analyzePattern(A);
285 // solver.factorize(A);
286 // xF = solver.solve(B);
287 // } else {
288  Eigen::BiCGSTAB<Eigen::SparseMatrix<double> > solver;
289  solver.compute(A);
290  xF = solver.solve(B);
291  std::cout << "#iterations: " << solver.iterations() << std::endl;
292  std::cout << "estimated error: " << solver.error() << std::endl;
293 // }
294  // Recover the cell unknowns
295  Eigen::VectorXd XTFh = Eigen::VectorXd::Zero(hho.ntotal_dofs());
296  XTFh.tail(hho.ntotal_edge_dofs()) = xF;
297 
298  if (hho.L()>=0) {
299  XTFh.head(hho.ntotal_cell_dofs()) = B_sc - A_sc_be * xF.head(hho.ntotal_edge_dofs());
300  } else {
301  XTFh.head(hho.ntotal_cell_dofs()) = A_sc_be * xF.head(hho.ntotal_edge_dofs());
302  }
303 
304  // Only Neumann: translate to get the proper average
305  if (BC==1){
306  // Compute average to translate
307  auto average = cell_quadrature.dot(XTFh.head(hho.ntotal_cell_dofs())) / total_measure;
308  auto average_exact_sol = hho.integrate_over_domain([&](auto x, auto y) {
309  return exact_solution(x,y);
310  }) / total_measure;
311 
312  // Translate the cells
313  for (size_t iT = 0; iT < mesh->n_cells(); iT++) {
314  XTFh(iT * hho.nlocal_cell_dofs()) += average_exact_sol - average;
315  }
316 
317  for (size_t iF = 0; iF < mesh->n_edges(); iF++) {
318  XTFh(hho.ntotal_cell_dofs() + iF * hho.nlocal_edge_dofs()) += average_exact_sol - average;
319  }
320  }
321 
322  solving_time = timer.elapsed(); // Record the final solving time
323 
324  // Compute the error in the linear system
325  solving_error = (A * xF - B).norm();
326 
327  return XTFh;
328 }

334 Eigen::MatrixXd LocVarDiff::diffusion_operator(HybridCore &hho, const size_t iT) const {
335 
336  boost::timer timeint;
337 
338  const auto mesh = hho.get_mesh_ptr();
339  const size_t dimPKcell = hho.dim_Pcell(hho.K());
340  const size_t dimPKcell_vec = mesh->dim() * dimPKcell;
341  const size_t nedgesT = mesh->cell(iT)->n_edges();
342 
343  // Total number of degrees of freedom local to this cell (cell and its adjacent faces)
344  size_t local_dofs = hho.nlocal_cell_dofs() + nedgesT * hho.nlocal_edge_dofs();
345 
346  //------------------- Initialisatons: quadratures, mass matrices... --------------------//
347 
348 //timeint.restart();
349 //itime[0] += timeint.elapsed();
350 //timeint.restart();
351 
352  // Note: representing the gradient reconstruction GT supposes a basis for (P^k)^d. This basis can be
353  // built from the basis (phi_i)_i of P^k, by working component-by-component:
354  // (phi_1 e_1, ..., phi_N e_1, phi_1 e_2, ..., phi_N e_2, ...),
355  // where e_1=[1 0], e_2=[0 1]... is the canonical basis of R^d
356  // Hence, the first dimPKcell functions are on the first component of R^d, the next dimPKcell on the second
357  // component etc.
358  // We do not explicitly build this basis function, but we compute the values of these functions at the
359  // quadrature points, using the values computed for the scalar basis functions
360 
361  // Diffusion in the cell.
362  tensor_function_type kappaT = [&](double x, double y){
363  // Does not work if kappa has a discontinuity on an edge
364  return kappa(x, y);
365  };
366 
367  // QUADRATURES
368  // Cell quadrature points, and values of cell basis functions (up to degree K+1), gradients of them,
369  // and vector basis functions (up to degree K) at these points
370  size_t doeT = std::max(2*(hho.K()+1), 2*hho.K() + _deg_kappa);
371  std::vector<HybridCore::qrule> quadT = hho.cell_qrule(iT, doeT);
372  std::vector<Eigen::VectorXd> phiT_quadT = hho.basis_quad('T', iT, quadT, hho.nhighorder_dofs());
373  std::vector<Eigen::MatrixXd> dphiT_quadT = hho.grad_basis_quad(iT, quadT, hho.nhighorder_dofs());
374  std::vector<Eigen::MatrixXd> vec_phiT_quadT(dimPKcell_vec, Eigen::MatrixXd::Zero(mesh->dim(), quadT.size()));
375  for (size_t r=0; r < mesh->dim(); r++){
376  for (size_t i=0; i < dimPKcell; i++){
377  vec_phiT_quadT[r * dimPKcell + i].row(r) = phiT_quadT[i].transpose();
378  }
379  }
380 
381  // MASS MATRICES
382  // Scalar cell mass matrix (phi_i,phi_j)_T up to degree max(L,K) * (K+1), and
383  // vector cell mass matrix (Phi_i,Phi_j)_T up to degree K*K [this one is block diagonal]
384  size_t maxdimPKL = std::max(hho.nlocal_cell_dofs(), dimPKcell);
385  Eigen::MatrixXd MTT = hho.gram_matrix(phiT_quadT, phiT_quadT, maxdimPKL, hho.nhighorder_dofs(), quadT, true);
386  Eigen::MatrixXd VecMTT = Eigen::MatrixXd::Zero(dimPKcell_vec, dimPKcell_vec);
387  for (size_t r=0; r < mesh->dim(); r++){
388  VecMTT.block(r*dimPKcell, r*dimPKcell, dimPKcell, dimPKcell) = MTT.topLeftCorner(dimPKcell, dimPKcell);
389  }
390 
391  // Face mass matrices:
392  // MFF[ilF]: face-face mass on face with local number ilF, up to degree K*K
393  // MFT[ilF]: face-cell mass on face with local number ilF, up to degree K*(K+1)
394  // MTT_on_F[ilF]: cell-cell mass on face with local number ilF, up to degree K*L
395  std::vector<Eigen::MatrixXd> MFF(nedgesT, Eigen::MatrixXd::Zero(hho.nlocal_edge_dofs(), hho.nlocal_edge_dofs()));
396  std::vector<Eigen::MatrixXd> MFT(nedgesT, Eigen::MatrixXd::Zero(hho.nlocal_edge_dofs(), hho.nhighorder_dofs()));
397  std::vector<Eigen::MatrixXd> MTT_on_F(nedgesT, Eigen::MatrixXd::Zero(dimPKcell, hho.nlocal_cell_dofs()));
398  for (size_t ilF = 0; ilF < nedgesT; ilF++) {
399  const size_t iF = mesh->cell(iT)->edge(ilF)->global_index();
400 
401  // Face quadrature points and values of cell and face basis functions (and gradients) at these points
402  size_t doeF = 2*hho.K() + 1;
403  std::vector<HybridCore::qrule> quadF = hho.edge_qrule(iF, doeF);
404  std::vector<Eigen::VectorXd> phiT_quadF = hho.basis_quad('T', iT, quadF, hho.nhighorder_dofs());
405  std::vector<Eigen::VectorXd> phiF_quadF = hho.basis_quad('F', iF, quadF, hho.nlocal_edge_dofs());
406  std::vector<Eigen::MatrixXd> dphiT_quadF = hho.grad_basis_quad(iT, quadF, hho.nhighorder_dofs());
407 
408  // Mass matrices
409  MFF[ilF] = hho.gram_matrix(phiF_quadF, phiF_quadF, hho.nlocal_edge_dofs(), hho.nlocal_edge_dofs(), quadF, true);
410  MFT[ilF] = hho.gram_matrix(phiF_quadF, phiT_quadF, hho.nlocal_edge_dofs(), hho.nhighorder_dofs(), quadF, false);
411  MTT_on_F[ilF] = hho.gram_matrix(phiT_quadF, phiT_quadF, dimPKcell, hho.nlocal_cell_dofs(), quadF, false);
412  }
413 
414  // STIFNESS mass-matrices (and the like): (\nabla phi_i,\nabla phi_j)_T up to degree (K+1)*(K+1)
415  // and (\nabla phi_i, Phi_j)_T up to degree (K+1)*K
416  Eigen::MatrixXd StiffT = hho.gram_matrix(dphiT_quadT, dphiT_quadT, hho.nhighorder_dofs(), hho.nhighorder_dofs(), quadT, true);
417  Eigen::MatrixXd MdphiT_PhiT = hho.gram_matrix(dphiT_quadT, vec_phiT_quadT, hho.nhighorder_dofs(), dimPKcell_vec, quadT, false);
418 
419 //itime[1] += timeint.elapsed();
420 //timeint.restart();
421 
422  //-------------------- Compute GT, matrix of full gradient reconstruction ---------//
423 
424  // Right-hand side, starting with volumetric term (Phi_i, \nabla phi_j)_T
425  Eigen::MatrixXd RHS_GT = Eigen::MatrixXd::Zero(dimPKcell_vec, local_dofs);
426  RHS_GT.topLeftCorner(dimPKcell_vec, hho.nlocal_cell_dofs()) = (MdphiT_PhiT.topLeftCorner(hho.nlocal_cell_dofs(), dimPKcell_vec)).transpose();
427 
428  // Boundary terms
429  for (size_t r=0; r < mesh->dim(); r++){
430  for (size_t ilF = 0; ilF < nedgesT; ilF++) {
431  // Offset for face unknowns
432  const size_t offset_F = hho.nlocal_cell_dofs() + ilF * hho.nlocal_edge_dofs();
433  const auto& nTF = mesh->cell(iT)->edge_normal(ilF);
434 
435  // Contribution of cell unknowns, and then face unknowns on F
436  RHS_GT.block(r*dimPKcell, 0, dimPKcell, hho.nlocal_cell_dofs()) -= nTF(r) * MTT_on_F[ilF];
437  RHS_GT.block(r*dimPKcell, offset_F, dimPKcell, hho.nlocal_edge_dofs()) +=
438  nTF(r) * (MFT[ilF].topLeftCorner(hho.nlocal_edge_dofs(), dimPKcell)).transpose();
439  }
440  }
441 
442  // Compute GT
443  Eigen::MatrixXd GT = (VecMTT.ldlt()).solve(RHS_GT);
444 
445  //------------- Consistent contribution (K GT, GT)_T to the local bilinear form ------//
446 
447  // Weighted mass matrix (K Phi_i, Phi_j)_T of the basis (Phi_i)_i of (P^k)^d
448  Eigen::MatrixXd kappaVecMTT = Eigen::MatrixXd::Zero(dimPKcell_vec, dimPKcell_vec);
449  kappaVecMTT = hho.gram_matrix(vec_phiT_quadT, vec_phiT_quadT, dimPKcell_vec, dimPKcell_vec, quadT, true, kappaT);
450 
451  Eigen::MatrixXd ATF = GT.transpose() * kappaVecMTT * GT;
452 
453  //------------- Compute PT, matrix of potential reconstruction, using GT ------//
454 
455  // We write that \nabla pT = projection on \nabla P^{k+1} of GT, and add the closure relation:
456  // (nabla pT v, nabla w)_T + lambda_T(p_T v,1)_T(w,1)_T = (GT v,\nabla w)_T + lambda_T(v_T,1)_T(w,1)_T
457 
458  // Right-hand side, starting with volumetric term
459  Eigen::MatrixXd RHS_PT = MdphiT_PhiT * GT;
460 
461  // Vector LT of (phi_j,1)_T for phi_j up to degree K+1, and LT^t*LT, for the closure relation
462  Eigen::VectorXd LT = (MTT.row(0)).transpose();
463  Eigen::MatrixXd LTtLT = LT * (LT.transpose());
464  double scalT = StiffT.trace() / std::pow(LT.norm(), 2);
465 
466  // Add closure relation and compute PT
467  RHS_PT.topLeftCorner(hho.nhighorder_dofs(), hho.nlocal_cell_dofs()) +=
468  scalT * LTtLT.topLeftCorner(hho.nhighorder_dofs(), hho.nlocal_cell_dofs());
469  Eigen::MatrixXd PT = ((StiffT + scalT*LTtLT).ldlt()).solve(RHS_PT);
470 
471 
472  //-------------------- Compute stabilisation term sT ---------//
473 
474  Eigen::MatrixXd STF = Eigen::MatrixXd::Zero(local_dofs, local_dofs);
475 
476  // Cell residual delta_T^l = pi_T^l (rT uT) - u_T
477 
478  Eigen::MatrixXd MTT_LKp1 = MTT.topLeftCorner(hho.nlocal_cell_dofs(), hho.nhighorder_dofs());
479  Eigen::MatrixXd MTT_LL = MTT.topLeftCorner(hho.nlocal_cell_dofs(), hho.nlocal_cell_dofs());
480  Eigen::MatrixXd deltaTL = MTT_LL.ldlt().solve( MTT_LKp1 * PT );
481  deltaTL.topLeftCorner(hho.nlocal_cell_dofs(), hho.nlocal_cell_dofs()) -=
482  Eigen::MatrixXd::Identity(hho.nlocal_cell_dofs(), hho.nlocal_cell_dofs());
483 
484  for (size_t ilF = 0; ilF < nedgesT; ilF++) {
485  auto hF = mesh->cell(iT)->edge(ilF)->measure();
486  auto xF = mesh->cell(iT)->edge(ilF)->center_mass();
487 
488  auto kappa_TF = kappaT(xF.x(), xF.y()).trace();
489 
490  // Face residual delta_TF^k = pi_F^k (rT uT) - u_F
491  Eigen::MatrixXd MFFinv = MFF[ilF].inverse();
492  Eigen::MatrixXd deltaTFK = MFFinv * MFT[ilF] * PT;
493  deltaTFK.block(0, hho.nlocal_cell_dofs() + ilF * hho.nlocal_edge_dofs(), hho.nlocal_edge_dofs(), hho.nlocal_edge_dofs()) -=
494  Eigen::MatrixXd::Identity(hho.nlocal_edge_dofs(), hho.nlocal_edge_dofs());
495 
496  // Stabilisation term
497  Eigen::MatrixXd dTFKminusdTL = deltaTFK - MFFinv * MFT[ilF].topLeftCorner(hho.nlocal_edge_dofs(), hho.nlocal_cell_dofs()) * deltaTL;
498 
499  STF += (kappa_TF / hF) * dTFKminusdTL.transpose() * MFF[ilF] * dTFKminusdTL;
500 
501  }
502 //itime[2] += timeint.elapsed();
503 //timeint.restart();
504 
505  // Adjust local bilinear form with stabilisation term
506  ATF += STF;
507 
508  return ATF;

517 Eigen::VectorXd LocVarDiff::load_operator(HybridCore &hho, const size_t iT) const {
518  // Load for the cell DOFs (first indices) and face DOFs (last indices)
519  const auto mesh = hho.get_mesh_ptr();
520  size_t cell_edge_dofs = hho.nlocal_cell_dofs() + mesh->cell(iT)->n_edges()*hho.nlocal_edge_dofs();
521  Eigen::VectorXd b = Eigen::VectorXd::Zero(cell_edge_dofs);
522 
523  // Quadrature points and values of cell basis functions at these points
524  std::vector<HybridCore::qrule> quadT = hho.cell_qrule(iT, 2*hho.K()+10);
525  size_t nbq = quadT.size();
526  std::vector<Eigen::VectorXd> phiT_quadT = hho.basis_quad('T', iT, quadT, hho.nlocal_cell_dofs());
527 
528  // Value of source times quadrature weights at the quadrature points
529  Eigen::VectorXd weight_source_quad = Eigen::VectorXd::Zero(nbq);
530  for (size_t iqn = 0; iqn < nbq; iqn++){
531  weight_source_quad(iqn) = quadT[iqn].w * source(quadT[iqn].x, quadT[iqn].y);
532  }
533 
534  for (size_t i=0; i < hho.nlocal_cell_dofs(); i++){
535  b(i) = weight_source_quad.dot(phiT_quadT[i]);
536  }
537  // Boundary values, if we have a boundary cell
538  if (mesh->cell(iT)->is_boundary()){
539  if (BC==0){
540  // Dirichlet BCs: no source terms on these edges
541  } else if (BC==1) {
542  // Neumann BCs
543  for (size_t ilF = 0; ilF < mesh->cell(iT)->n_edges(); ilF++) {
544  const size_t iF = mesh->cell(iT)->edge(ilF)->global_index();
545  // BC on boundary faces
546  if (mesh->cell(iT)->edge(ilF)->is_boundary()){
547  // Offset for face unknowns
548  const size_t offset_F = hho.nlocal_cell_dofs() + ilF * hho.nlocal_edge_dofs();
549  // Normal to the face
550  const auto& nTF = mesh->cell(iT)->edge_normal(ilF);
551  // for each DOF of the boundary face
552  for (size_t i = 0; i < hho.nlocal_edge_dofs(); i++){
553  const auto& phi_i = hho.edge_basis(iF, i);
554  b(offset_F + i) = hho.integrate_over_edge(iF, [&](double x, double y){
555  return nTF.dot(kappa(x,y) * grad_exact_solution(x,y)) * phi_i(x,y);
556  });
557  }
558  }
559  }
560  }
561  }
562 
563  return b;
564 }

566 double LocVarDiff::H1Norm(HybridCore& hho, const Eigen::VectorXd XTF) {
567  const auto mesh = hho.get_mesh_ptr();
568  double value = 0.0;
569  for (size_t iT = 0; iT < mesh-> n_cells(); iT++) {
570 
571  // Compute cell quadrature points and values of cell basis functions and gradients
572  std::vector<HybridCore::qrule> quadT = hho.cell_qrule(iT, 2*hho.Ldeg());
573  std::vector<Eigen::MatrixXd> dphiT_quadT = hho.grad_basis_quad(iT, quadT, hho.nlocal_cell_dofs());
574 
575  // CELL CONTRIBUTION
576  //
577  // Stiffness matrix
578  // (note: we compute also zeros first row and column...)
579  Eigen::MatrixXd MG = hho.gram_matrix(dphiT_quadT, dphiT_quadT, hho.nlocal_cell_dofs(), hho.nlocal_cell_dofs(), quadT, true);
580 
581  // Cell indices
582  Eigen::VectorXd XTF_T = XTF.segment(iT*hho.nlocal_cell_dofs(), hho.nlocal_cell_dofs());
583 
584  value += XTF_T.dot(MG * XTF_T);
585 
586  // Faces contributions
587 
588  for (size_t ilF=0; ilF < mesh->cell(iT)->n_edges() ; ilF++){
589  size_t iF = mesh->cell(iT)->edge(ilF)->global_index();
590  auto hF = mesh->edge(iF)->measure();
591  // Face quadrature points and values of cell and face basis functions at these points
592  std::vector<HybridCore::qrule> quadF = hho.edge_qrule(iF, 2*hho.K()+2);
593  std::vector<Eigen::VectorXd> phiT_quadF = hho.basis_quad('T', iT, quadF, hho.nlocal_cell_dofs());
594  std::vector<Eigen::VectorXd> phiF_quadF = hho.basis_quad('F', iF, quadF, hho.nlocal_edge_dofs());
595 
596  // Face, cell and face-cell Gram matrix on F
597  Eigen::MatrixXd MFF = hho.gram_matrix(phiF_quadF, phiF_quadF, hho.nlocal_edge_dofs(), hho.nlocal_edge_dofs(), quadF, true);
598  Eigen::MatrixXd MFT = hho.gram_matrix(phiF_quadF, phiT_quadF, hho.nlocal_edge_dofs(), hho.nlocal_cell_dofs(), quadF, false);
599  Eigen::MatrixXd MTT_on_F = hho.gram_matrix(phiT_quadF, phiT_quadF, hho.nlocal_cell_dofs(), hho.nlocal_cell_dofs(), quadF, true);
600 
601  // Face indices
602  Eigen::VectorXd XTF_F = XTF.segment(hho.ntotal_cell_dofs() + iF * hho.nlocal_edge_dofs(), hho.nlocal_edge_dofs());
603 
604  double valF = XTF_T.dot(MTT_on_F * XTF_T) - 2 * XTF_F.dot(MFT * XTF_T) + XTF_F.dot(MFF * XTF_F);
605 
606  value += valF / hF;
607  }
608  }
609  return sqrt(value);
610 
611 }

614 Eigen::VectorXd LocVarDiff::VertexValues(HybridCore& hho, const Eigen::VectorXd XTF, const char from_dofs) {
615  const auto mesh = hho.get_mesh_ptr();
616  Eigen::VectorXd function_vertex = Eigen::VectorXd::Zero(mesh->n_vertices());
617 
618  if (from_dofs == 'T'){
619  // compute from the cell polynomials, averaging all values coming from the cells around each vertex
620  for (size_t iV = 0; iV < mesh->n_vertices(); iV++){
621  auto xV = mesh->vertex(iV)->coords();
622  auto cList = mesh->vertex(iV)->get_cells();
623  auto weight = cList.size();
624  for (size_t ilT = 0; ilT < weight; ilT++){
625  auto temp = hho.evaluate_in_cell(XTF, cList[ilT]->global_index(), xV.x(), xV.y());
626  function_vertex(iV) += temp;
627  }
628  function_vertex(iV) = function_vertex(iV)/(weight);
629  }
630 
631  }else{
632  // compute from the edge polynomials, averaging all values coming from the edges around each vertex
633  for (size_t iV = 0; iV < mesh->n_vertices(); iV++){
634  auto xV = mesh->vertex(iV)->coords();
635  auto eList = mesh->vertex(iV)->get_edges();
636  double weight = eList.size();
637  for (size_t ilF = 0; ilF < weight; ilF++){
638  auto temp = hho.evaluate_in_edge(XTF, eList[ilF]->global_index(), xV.x(), xV.y());
639  function_vertex(iV) += temp;
640  }
641  function_vertex(iV) = function_vertex(iV)/(weight);
642  }
643  }
644 
645  return function_vertex;
